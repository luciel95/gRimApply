#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    int column = board[0].size();
    int row = board.size();
    
    for (int j = 0; j < column; j ++)
    {
        for (int i = 0; i < row; i++)
        {
            if ((i + j) <= k)
            {
                answer += board[i][j];
            }
        }
    }
    
    return answer;
}
